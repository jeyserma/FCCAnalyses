41
42 auto c = TMVA::Experimental::Internal::ParseXMLConfig(filename);

46 fNumClasses = c.numClasses;

49 const auto numVars = fVariables.size();
50 fValues = std::vector<float>(numVars);
51
52 for(std::size_t i = 0; i < numVars; i++) {
53 fReader->AddVariable(TString(fExpressions[i]), &fValues[i]);
54 }
55 fReader->BookMVA(name, filename.c_str());
56
57 }
58
59 std::vector<float> Compute(const Vec_f &x) {
60
61 if (x.size() != fVariables.size())
62 throw std::runtime_error("Size of input vector is not equal to number of variables.");
63
64 // Copy over inputs to memory used by TMVA reader
65 for (std::size_t i = 0; i < x.size(); i++) {
66 fValues[i] = x[i];
67 }
68
69 // Evaluate TMVA model
70 // Classification
71 if (fAnalysisType == TMVA::Experimental::Internal::AnalysisType::Classification) {
72 return std::vector<float>({static_cast<float>(fReader->EvaluateMVA(name))});
73 }
74 // Regression
75 else if (fAnalysisType == TMVA::Experimental::Internal::AnalysisType::Regression) {
76 return fReader->EvaluateRegression(name);
77 }
78 // Multiclass
79 else if (fAnalysisType == TMVA::Experimental::Internal::AnalysisType::Multiclass) {
80 return fReader->EvaluateMulticlass(name);
81 }
82 // Throw error
83 else {
84 throw std::runtime_error("RReader has undefined analysis type.");
85 return std::vector<float>();
86 }
87 }
88
89 private:
90 std::unique_ptr<TMVA::Reader> fReader;
91 std::vector<float> fValues;
92 std::vector<std::string> fVariables;
93 std::vector<std::string> fExpressions;
94 unsigned int fNumClasses;
